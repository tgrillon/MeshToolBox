86void Application::Run()
87{
88 m_Running = true;
89
90 float lastTime = GetTime();
91
92 // Main Application loop
93 while(m_Running)
94 {
95 glfwPollEvents();
96
97 if(m_Window->ShouldClose())
98 {
99 Stop();
100 break;
101 }
102
103 float currentTime = GetTime();
104 float timestep = glm::clamp(currentTime - lastTime, 0.001f, 0.1f);
105 lastTime = currentTime;
106
107 // Main layer update here
108 for(const std::unique_ptr<Layer>& layer : m_LayerStack)
109 layer->OnUpdate(timestep);
110
111 // Start the Dear ImGui frame
112 ImGui_ImplOpenGL3_NewFrame();
113 ImGui_ImplGlfw_NewFrame();
114 ImGui::NewFrame();
115
116 {
117 static ImGuiDockNodeFlags dockspace_flags = ImGuiDockNodeFlags_None;
118
119 // We are using the ImGuiWindowFlags_NoDocking flag to make the parent window not dockable into,
120 // because it would be confusing to have two docking targets within each others.
121 ImGuiWindowFlags window_flags = ImGuiWindowFlags_NoDocking;
122 if(m_MenubarCallback)
123 window_flags |= ImGuiWindowFlags_MenuBar;
124
125 const ImGuiViewport* viewport = ImGui::GetMainViewport();
126 ImGui::SetNextWindowPos(viewport->WorkPos);
127 ImGui::SetNextWindowSize(viewport->WorkSize);
128 ImGui::SetNextWindowViewport(viewport->ID);
129 ImGui::PushStyleVar(ImGuiStyleVar_WindowRounding, 0.0f);
130 ImGui::PushStyleVar(ImGuiStyleVar_WindowBorderSize, 0.0f);
131 window_flags |= ImGuiWindowFlags_NoTitleBar | ImGuiWindowFlags_NoCollapse | ImGuiWindowFlags_NoResize
132 | ImGuiWindowFlags_NoMove;
133 window_flags |= ImGuiWindowFlags_NoBringToFrontOnFocus | ImGuiWindowFlags_NoNavFocus;
134
135 // When using ImGuiDockNodeFlags_PassthruCentralNode, DockSpace() will render our background
136 // and handle the pass-thru hole, so we ask Begin() to not render a background.
137 if(dockspace_flags & ImGuiDockNodeFlags_PassthruCentralNode)
138 window_flags |= ImGuiWindowFlags_NoBackground;
139
140 // Important: note that we proceed even if Begin() returns false (aka window is collapsed).
141 // This is because we want to keep our DockSpace() active. If a DockSpace() is inactive,
142 // all active windows docked into it will lose their parent and become undocked.
143 // We cannot preserve the docking relationship between an active window and an inactive docking, otherwise
144 // any change of dockspace/settings would lead to windows being stuck in limbo and never being visible.
145 ImGui::PushStyleVar(ImGuiStyleVar_WindowPadding, ImVec2(0.0f, 0.0f));
146 ImGui::Begin("DockSpace Demo", nullptr, window_flags);
147 ImGui::PopStyleVar();
148
149 ImGui::PopStyleVar(2);
150
151 // Submit the DockSpace
152 ImGuiIO& io = ImGui::GetIO();
153 if(io.ConfigFlags & ImGuiConfigFlags_DockingEnable)
154 {
155 ImGuiID dockspace_id = ImGui::GetID("VulkanAppDockspace");
156 ImGui::DockSpace(dockspace_id, ImVec2(0.0f, 0.0f), dockspace_flags);
157 }
158
159 if(m_MenubarCallback)
160 {
161 if(ImGui::BeginMenuBar())
162 {
163 m_MenubarCallback();
164 ImGui::EndMenuBar();
165 }
166 }
167
168 // NOTE: rendering can be done elsewhere (eg. render thread)
169 for(const std::unique_ptr<Layer>& layer : m_LayerStack)
170 layer->OnRender();
171
172 ImGui::End();
173 }
174
175 ImGui::Render();
176 glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
177
178 ImGuiIO& io = ImGui::GetIO();
179 IM_UNUSED(io);
180 ImGui_ImplOpenGL3_RenderDrawData(ImGui::GetDrawData());
181 if(io.ConfigFlags & ImGuiConfigFlags_ViewportsEnable)
182 {
183 GLFWwindow* backup_current_context = glfwGetCurrentContext();
184 ImGui::UpdatePlatformWindows();
185 ImGui::RenderPlatformWindowsDefault();
