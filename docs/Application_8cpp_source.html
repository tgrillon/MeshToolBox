92void Application::Run()
93{
94 m_Running = true;
95
96 float lastTime = GetTime();
97
98 // Main Application loop
99 while(m_Running)
100 {
101 glfwPollEvents();
102
103 if(m_Window->ShouldClose())
104 {
105 Stop();
106 break;
107 }
108
109 float currentTime = GetTime();
110 float timestep = glm::clamp(currentTime - lastTime, 0.001f, 0.1f);
111 lastTime = currentTime;
112
113 // Main layer update here
114 for(const std::unique_ptr<Layer>& layer : m_LayerStack)
115 layer->OnUpdate(timestep);
116
117 // Start the Dear ImGui frame
118 ImGui_ImplOpenGL3_NewFrame();
119 ImGui_ImplGlfw_NewFrame();
120 ImGui::NewFrame();
121
122 {
123 static ImGuiDockNodeFlags dockspace_flags = ImGuiDockNodeFlags_None;
124
125 // We are using the ImGuiWindowFlags_NoDocking flag to make the parent window not dockable into,
126 // because it would be confusing to have two docking targets within each others.
127 ImGuiWindowFlags window_flags = ImGuiWindowFlags_NoDocking;
128 if(m_MenubarCallback)
129 window_flags |= ImGuiWindowFlags_MenuBar;
130
131 const ImGuiViewport* viewport = ImGui::GetMainViewport();
132 ImGui::SetNextWindowPos(viewport->WorkPos);
133 ImGui::SetNextWindowSize(viewport->WorkSize);
134 ImGui::SetNextWindowViewport(viewport->ID);
135 ImGui::PushStyleVar(ImGuiStyleVar_WindowRounding, 0.0f);
136 ImGui::PushStyleVar(ImGuiStyleVar_WindowBorderSize, 0.0f);
137 window_flags |= ImGuiWindowFlags_NoTitleBar | ImGuiWindowFlags_NoCollapse | ImGuiWindowFlags_NoResize
138 | ImGuiWindowFlags_NoMove;
139 window_flags |= ImGuiWindowFlags_NoBringToFrontOnFocus | ImGuiWindowFlags_NoNavFocus;
140
141 // When using ImGuiDockNodeFlags_PassthruCentralNode, DockSpace() will render our background
142 // and handle the pass-thru hole, so we ask Begin() to not render a background.
143 if(dockspace_flags & ImGuiDockNodeFlags_PassthruCentralNode)
144 window_flags |= ImGuiWindowFlags_NoBackground;
145
146 // Important: note that we proceed even if Begin() returns false (aka window is collapsed).
147 // This is because we want to keep our DockSpace() active. If a DockSpace() is inactive,
148 // all active windows docked into it will lose their parent and become undocked.
149 // We cannot preserve the docking relationship between an active window and an inactive docking, otherwise
150 // any change of dockspace/settings would lead to windows being stuck in limbo and never being visible.
151 ImGui::PushStyleVar(ImGuiStyleVar_WindowPadding, ImVec2(0.0f, 0.0f));
152 ImGui::Begin("DockSpace Demo", nullptr, window_flags);
153 ImGui::PopStyleVar();
154
155 ImGui::PopStyleVar(2);
156
157 // Submit the DockSpace
158 ImGuiIO& io = ImGui::GetIO();
159 if(io.ConfigFlags & ImGuiConfigFlags_DockingEnable)
160 {
161 ImGuiID dockspace_id = ImGui::GetID("VulkanAppDockspace");
162 ImGui::DockSpace(dockspace_id, ImVec2(0.0f, 0.0f), dockspace_flags);
163 }
164
165 if(m_MenubarCallback)
166 {
167 if(ImGui::BeginMenuBar())
168 {
169 m_MenubarCallback();
170 ImGui::EndMenuBar();
171 }
172 }
173
174 // NOTE: rendering can be done elsewhere (eg. render thread)
175 for(const std::unique_ptr<Layer>& layer : m_LayerStack)
176 layer->OnRender();
177
178 ImGui::End();
179 }
180
181 ImGui::Render();
182 glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
183
184 ImGuiIO& io = ImGui::GetIO();
185 IM_UNUSED(io);
186 ImGui_ImplOpenGL3_RenderDrawData(ImGui::GetDrawData());
187 if(io.ConfigFlags & ImGuiConfigFlags_ViewportsEnable)
188 {
189 GLFWwindow* backup_current_context = glfwGetCurrentContext();
190 ImGui::UpdatePlatformWindows();
191 ImGui::RenderPlatformWindowsDefault();
192 glfwMakeContextCurrent(backup_current_context);
193 }
194
195 m_Window->Update();
196 }
